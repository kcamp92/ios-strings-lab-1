## Question 1

Write code that prints out all the numbers from 1 to 10 as a single string.
(Hint: the `String()` function can convert an Int to a String)

var emptyString = ""
let int = 1...10
for a in int {
emptyString += String(a)
}
print(emptyString)

emptyString += " \(a) "
}

print(emptyString)

emptyString += "\(initialNumber) "
}
